# Assuming all your data frames have the same number of rows
N <- nrow(nest_density)

# Assuming you have a column for peninsula in one of your datasets
peninsula <- as.numeric(factor(nest_density$peninsula))
P <- length(unique(peninsula))

# Assuming you have a column for year in one of your datasets
year <- as.numeric(factor(nest_density$year))
N_years <- length(unique(year))

bugs_data <- list(
  N = N,
  P = P,
  y = colony_growth_index$growth_index,
  nest_density = nest_density$density,
  road_proximity = road_proximity$proximity,
  bcnh_nest_success = bcnh_nest_success$success,
  dcco_usurpation = dcco_usurpation$usurpation,
  dcco_management = dcco_management$management,
  peninsula = peninsula,
  N_years = N_years,
  year = year
)

# Initial values for the MCMC chains
inits <- function() {
  list(
    beta = rnorm(6, 0, 1),
    sigma = runif(1, 0, 10),
    sigma_peninsula = runif(1, 0, 10),
    sigma_year = runif(1, 0, 10),
    alpha_peninsula = rnorm(P, 0, 1),
    alpha_year = rnorm(N_years, 0, 1)
  )
}
